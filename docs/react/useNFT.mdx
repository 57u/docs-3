Hook for fetching an NFT token from an NFT Contract.

```jsx
import { useNFT } from "@thirdweb-dev/react"
```

### Usage

:::tip

It is recommended to use `useContract` to initialize your NFT contract before using this hook

```jsx
import { useContract } from "@thirdweb-dev/react";
const { contract, isLoading, error } = useContract("{{nft_contract_address}}", "nft-contract-type");
```
:::


```jsx 
const { contract } = useContract(<ContractAddress>);
const { data: nft, isLoading, error } = useNFT(contract, <tokenId>);

if (isLoading) return <div>Fetching NFTâ€¦</div>
if (isError) return <div>Error fetching token</div>
return <div>NFT: {data?.metadata.title}</div>
```

error?: Error
isIdle: boolean
isLoading: boolean
isFetching: boolean
isSuccess: boolean
isError: boolean
isFetched: boolean
isFetchedAfterMount: boolean
