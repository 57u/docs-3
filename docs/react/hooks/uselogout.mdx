Hook for logging out a connected wallet.

```jsx
import { useLogout } from "@thirdweb-dev/react";
```

## Usage

```jsx
import { useLogout, Web3Button } from "@thirdweb-dev/react";

function App() {
  const { logout, isLoading } = useLogout();

  return (
    <Web3Button action={() => logout()} disabled={isLoading}>
      Logout
    </Web3Button>
  );
}
```
