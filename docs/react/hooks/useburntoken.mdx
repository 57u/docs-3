Hook for burning ERC20 tokens on a smart contract.

Available to use on smart contracts that implement the [ERC20](/contractkit/interfaces/erc20) standard.

```jsx
import { useBurnToken } from "@thirdweb-dev/react";
```

## Usage

Provide your token contract as the argument.

```jsx
import { useBurnToken, useContract, Web3Button } from "@thirdweb-dev/react";

// Your smart contract address
const contractAddress = "{{contract_address}}";

function App() {
  const { contract } = useContract(contractAddress);
  const { mutateAsync: burnToken, isLoading, error } = useBurnToken(contract);

  return (
    <Web3Button
      contractAddress={contractAddress}
      action={() =>
        burnToken({
          to: "{{wallet_address}}",
          amount: "10", // Amount of tokens to burn
        })
      }
    >
      Burn Token
    </Web3Button>
  );
}
```

## Configuration

### to

Likely, you will want to burn the token from the currently connected wallet address.

You can use the [`useAddress`](/react/hooks/useAddress) hook to get this value.

```jsx
import {
  useBurnToken,
  useContract,
  Web3Button,
  useAddress,
} from "@thirdweb-dev/react";

// Your smart contract address
const contractAddress = "{{contract_address}}";

function App() {
  // highlight-next-line
  const address = useAddress();
  const { contract } = useContract(contractAddress);
  const { mutateAsync: burnToken, isLoading, error } = useBurnToken(contract);

  return (
    <Web3Button
      contractAddress={contractAddress}
      action={() =>
        burnToken({
          // highlight-next-line
          to: address,
          amount: "10", // Amount of tokens to burn
        })
      }
    >
      Burn Token
    </Web3Button>
  );
}
```
