Hook for burning a token on a smart contract.

Available to use on smart contracts that implement the [ERC20](/contractkit/interfaces/erc20)
or [ERC721](/contractkit/interfaces/erc721) standard.

```jsx
import { useBurnToken } from "@thirdweb-dev/react";
```

## Usage

Provide your token contract as the argument.

```jsx
import { useBurnToken, useContract, Web3Button } from "@thirdweb-dev/react";

// Your smart contract address
const contractAddress = "{{contract_address}}";

function App() {
  const { contract } = useContract(contractAddress);
  const { mutateAsync: burnToken, isLoading, error } = useBurnToken(contract);

  if (error) {
    console.error("failed to burn token", error);
  }

  return (
    <Web3Button
      contractAddress={contractAddress}
      action={() =>
        burnToken({
          to: "{{wallet_address}}", // Use useAddress hook to get current wallet address
          amount: "10", // Amount of tokens to burn
        })
      }
    >
      Burn Token
    </Web3Button>
  );
}
```

## Configuration

### to

Likely, you will want to burn the token from the currently connected wallet address.

You can use the [`useAddress`](/react/hooks/useAddress) hook to get this value.

```jsx
import {
  useBurnToken,
  useContract,
  Web3Button,
  useAddress,
} from "@thirdweb-dev/react";

// Your smart contract address
const contractAddress = "{{contract_address}}";

function App() {
  // highlight-start
  const address = useAddress();
  // highlight-end
  const { contract } = useContract(contractAddress);
  const { mutateAsync: burnToken, isLoading, error } = useBurnToken(contract);

  if (error) {
    console.error("failed to burn token", error);
  }

  return (
    <Web3Button
      contractAddress={contractAddress}
      action={() =>
        burnToken({
          // highlight-start
          to: address, // Use useAddress hook to get current wallet address
          // highlight-end
          amount: "10", // Amount of tokens to burn
        })
      }
    >
      Burn Token
    </Web3Button>
  );
}
```
