Hook for creating an auction listing on a marketplace smart contract.

Available to use on smart contracts that implement the [Marketplace](/contractkit/interfaces/marketplace)
or [MarketplaceV3](/contractkit/interfaces/marketplacev3) standard.

```jsx
import { useCreateAuctionListing } from "@thirdweb-dev/react";
```

## Usage

Provide your [Marketplace](https://thirdweb.com/thirdweb.eth/Marketplace)
or
[MarketplaceV3](https://thirdweb.com/thirdweb.eth/MarketplaceV3)
contract as the argument to the hook.

```jsx
import {
  useCreateAuctionListing,
  useContract,
  Web3Button,
} from "@thirdweb-dev/react";

// Your smart contract address
const contractAddress = "{{contract_address}}";

function App() {
  const { contract } = useContract(contractAddress, "marketplace");
  const {
    mutateAsync: createAuctionListing,
    isLoading,
    error,
  } = useCreateAuctionListing(contract);

  return (
    <Web3Button
      contractAddress={contractAddress}
      action={() =>
        createAuctionListing({
          tokenId: "{{token_id}}", // The ID of the token to list.
          assetContractAddress: "{{asset_contract_address}}", // The contract address of the asset being listed.
          buyoutPricePerToken: "{{buyout_price_per_token}}", // The buyout price per token on the listing.
          currencyContractAddress: "{{currency_contract_address}}", // The address of the currency to accept for the listing.
          listingDurationInSeconds: "{{listing_duration_in_seconds}}", // The duration of the listing in seconds.
          quantity: "{{quantity}}", // The quantity of tokens to include in the listing. (always 1 for ERC721)
          reservePricePerToken: "{{reserve_price_per_token}}", // Minimum price that a bid must be in order to be accepted, per token.
          startTimestamp: new Date(), // The start time of the listing.
        })
      }
    >
      Create Auction Listing
    </Web3Button>
  );
}
```

### buyoutPricePerToken

- For ERC721 NFTs, this is the price to buy the NFT outright.

### currencyContractAddress

The address of the currency you want users to pay with and make bids in.

You likely want to use the token native to the chain you are on, e.g. Ether on Ethereum.

To do that, you can import the `NATIVE_TOKEN_ADDRESS` constant from `@thirdweb-dev/sdk`.

```jsx

  useCreateAuctionListing,
  useContract,
  Web3Button,
} from "@thirdweb-dev/react";
// highlight-next-line
import { NATIVE_TOKEN_ADDRESS } from "@thirdweb-dev/sdk";

// Your smart contract address
const contractAddress = "{{contract_address}}";

  const { contract } = useContract(contractAddress, "marketplace");
  const {
    mutateAsync: createAuctionListing,
    isLoading,
    error,
  } = useCreateAuctionListing(contract);

  return (
    <Web3Button
      contractAddress={contractAddress}
      action={() =>
        createAuctionListing({
          tokenId: "{{token_id}}", // The ID of the token to list.

          buyoutPricePerToken: "{{buyout_price_per_token}}", // The buyout price per token on the listing.
          // highlight-next-line
          currencyContractAddress: NATIVE_TOKEN_ADDRESS, // The address of the currency to accept for the listing.
          listingDurationInSeconds: "{{listing_duration_in_seconds}}", // The duration of the listing in seconds.
          quantity: "{{quantity}}", // The quantity of tokens to include in the listing. (always 1 for ERC721)

  );
}
```
